249std::vector<std::vector<double>> sc::mzxml::MZXML::extract_spectrum(const pugi::xml_node& spectrum_node) {
250
251 std::vector<std::vector<double>> spectrum(2);
252
253 int number_traces = spectrum_node.attribute("peaksCount").as_int();
254
255 if (number_traces == 0) return spectrum;
256
257 for (int i = 0; i < 2; i++) spectrum[i].resize(number_traces);
258
259 const char* encoded_string = spectrum_node.child("peaks").child_value();
260
261 std::string decoded_string = utils::decode_base64(encoded_string);
262
263 if (binary_metadata.compressed) {

265 }
266
267 std::vector<double> res(number_traces * 2);
268
269 if (binary_metadata.byte_order == "big_endian") {

272 } else if (binary_metadata.byte_order == "little_endian") {
273 res = utils::decode_little_endian(decoded_string, binary_metadata.precision / 8);
