5This script investigates the optimal window size and position according to the regression error to compare the performance of the different regressions used to describe the same peak.

8def regression(scale, y):
9 """
10 This function calculates the regression error of a gaussian fit to the data.
11 scale: window size
12 y: data
13 """
14 # create design Matrix
15 x = np.linspace(-scale, scale, 2*scale +1) # x values for the data
16 X = np.c_[np.ones(len(x)), x, x**2, x**2] # design matrix
17 X[x<0, 3] = 0 # set negative values to 0
18 X[x>0, 2] = 0 # set positive values to 0
19 b = np.linalg.inv(X.T @ X) @ X.T @ np.log(y) # regression coefficients
20 y_pred = np.exp(X @ b) # predicted y values
21
22 """add a weight to the error based on the distance to the regression apex"""
23 # calculate the apex of the regression
24 if b[1] < 0:
25 apex = -b[1] / (2*b[2])
26 else:
27 apex = -b[1] / (2*b[3])
28
29 # calculate the distance to the apex
30 distance = np.abs(x - apex)
31
32 # calculate the weight thereby a distance higher than scale should be almost 0
33 weight = np.exp(-.5 * (distance / scale)**2)
34 # normalize the weight
35 weight /= np.sum(weight)
36
37 error = np.sum((y - y_pred)**2 * weight) # regression error
38 mse = error
39 return mse
