8 void MeasurementData::zeroFilling(varDataType &dataVec, int k)
9 {
10 std::visit([k](auto &&arg)
11 {
12 // iterate over the vector of varDataType datatype objects
13 for (auto &pair : *arg)
14 {
15 // de-reference the unique pointer of the object
16 auto &dataObj = *(pair.get());
17 // initialize the expected difference
18 double expectedDifference = 0.0;
19 // initialize the vector of differences
20 std::vector<double> differences;
21 // iterate over data point vector, which is a vector of unique pointers to data points structures
22 auto &data = dataObj.dataPoints;
23
24 double previousX = data[0]->x();
25 for (int i = 1; i < data.size(); i++)
26 {
27 double difference = data[i]->x() - previousX;
28 differences.push_back(difference);
29 previousX = data[i]->x();
30 }
31
32 // median of differences
33 // copy the differences vector and sort it
34 std::vector<double> differences_copy = differences;
35 std::sort(differences_copy.begin(), differences_copy.end());
36 int n = differences_copy.size();
37 if (n % 2 == 0)
38 {
39 expectedDifference = (differences_copy[n / 2 - 1] + differences_copy[n / 2]) / 2;
40 }
41 else
42 {
43 expectedDifference = differences_copy[n / 2];
44 }
45 // delete the copy of the differences vector
46 std::vector<double>().swap(differences_copy);
47
48 /* analyze the data for gaps, i.e., differences > 1.75 * expectedDifference, and fill the gaps by adding new data points at the end of the vector
49 */
50 n++; // adjust n to fit with the size of the data vector
51 size_t counter = k / 2; // counter for the number of added data points to store separator positions. it is initialized with k/2 to address that later k/2 data points will be added at the front of the vector
52 std::vector<size_t> separators;
53 for (int i = 1; i < n; i++)
54 {
55 // consider the difference between two neighboring data points from differences vector and compare it with 1.75 * expectedDifference
56 if (differences[i - 1] > 1.75 * expectedDifference)
57 {
58 // calculate the gap size; need to be rounded to the nearest integer
59 int gapSize = std::round(differences[i - 1] / expectedDifference - 1);
60 // check if the gapSize is larger than k, as this is the maximum gap size
61 if (gapSize <= k)
62 {
63 // add gapSize new data points at the end of the vector
64 for (int j = 1; j <= gapSize; j++)
65 {
66 // define the new pointer for new data point and add it to the data vector using the addDataPoint method from the respecting class
67 dataObj.addDataPoint(0.0, data[i - 1]->x() + j * expectedDifference, 0);
68 }
69 counter += gapSize;
70 }
71 else
72 {
73 // limit the gap size to k and add k/2 new data points close to the (i-1) data point and k/2 new data points close to the (i) data point
74 gapSize = k;
75 // add k/2 new data points close to the (i) and (i-1) data point
76 for (int j = 1; j <= k / 2; j++)
77 {
78 // define the new pointer for new data point and add it to the data vector
79 dataObj.addDataPoint(0.0, data[i]->x() - j * expectedDifference, 0);
80 dataObj.addDataPoint(0.0, data[i - 1]->x() + j * expectedDifference, 0);
81 }
82 counter += k / 2;
83 // add the separator to the vector
84 separators.push_back(counter);
85 counter += k / 2;
86 }
87 }
88 else
89 {
90 // update expectedDifference
91 expectedDifference = expectedDifference * .5 + differences[i - 1] * .5;
92 }
93 counter++;
94 }
95 // extrapolate the data by adding k/2 new data points at the end of the vector
96 int gapSize = (int)k / 2;
97 for (int j = 1; j <= gapSize; j++)
98 {
99 // define the new pointer for new data point and add it to the data vector
100 dataObj.addDataPoint(0.0, data[n - 1]->x() + j * expectedDifference, 0);
101 dataObj.addDataPoint(0.0, data[0]->x() - j * expectedDifference, 0);
102 }
103 // add the last index of data vector to the separators vector
104 separators.push_back(dataObj.dataPoints.size() - 1);
105 // flip separators vector
106 std::reverse(separators.begin(), separators.end());
107 // cumulative difference for separators // HERE IS SOME ERROR
108 for (int i = 0; i < separators.size() - 1; i++)
109 {
110 separators[i] = separators[i] - separators[i + 1];
111 }
112 // delete the last element of the separators vector
113 separators.pop_back();
114 // add the separators to the Object's cuttingPoints vector, which is a vector of unique pointers to size_t objects
115 for (int i = 0; i < separators.size(); i++)
116 {
117 dataObj.cuttingPoints.push_back(std::make_unique<size_t>(separators[i]));
118 }
119 // sort the data points by x-axis values
120 dataObj.sortDataPoints();
121 } // end of for loop
122 },
123 dataVec); // end of visit
124 } // end of zeroFilling

205 {
206 std::vector<int> keysToDelete;

216 // ALL THE DATA SETS CONSIDERED HAVE AT LEAST 4 DATA POINTS WITH INTENSITY > 0 (due to filterSmallDataSets)
217 // extract the largest y value, which is used for thershoulding the extrapolations, i.e. MAX < 3 * max(y)
218 double MAX = 0.0;

223 MAX = dataPoints[i]->y();
224 }

227 /* Data points at the edges will be extrapolated. Therefore, first and last non-zero data points and maximum data point are log-transformed and considered for parabola definition. Edge data points are then replaced by the extrapolated values from parabola and back transformed into linear space using exp function.*/
228 // find the first and last data points in dataPoints->y() that are not zero and store them in xDataTemp and yDataTemp
229 std::vector<double> xDataTemp;
230 std::vector<double> yDataTemp;
231 int I = 0;
232 int J = 0;
233 // iterate over the data points to find the first data point that is not zero
234 for (int i = 0; i < dataPoints.size(); i++)
235 {
236 if (dataPoints[i]->y() > 0.0)
237 {
238 I = i;
239 xDataTemp.push_back(dataPoints[i]->x() - dataPoints[0]->x());
240 yDataTemp.push_back(log(dataPoints[i]->y())); // we use log space for extrapolation
241 break;
242 }
243 }
244 // iterate over the data points to find the last data point that is not zero
245 for (int j = dataPoints.size() - 1; j >= 0; j--)
246 {
247 if (dataPoints[j]->y() > 0.0)
248 {
249 J = j;
250 xDataTemp.push_back(dataPoints[j]->x() - dataPoints[0]->x());
251 yDataTemp.push_back(log(dataPoints[j]->y())); // we use log space for extrapolation
252 break;
253 }
254 }
255 // find the maximum data point
256 double max = 0.0;
257 int maxIndex = 0;
258 for (int k = I; k < J; k++)
259 {
260 if (dataPoints[k]->y() > max)
261 {
262 max = dataPoints[k]->y();
263 maxIndex = k;
264 }
265 }
266 xDataTemp.push_back(dataPoints[maxIndex]->x() - dataPoints[0]->x());
267 yDataTemp.push_back(log(dataPoints[maxIndex]->y())); // we use log space for extrapolation
268
269 /* For later peak detection, it has to be ensured that the maximum of y values is not the first or last non-zero data point.*/
270 if (maxIndex == I || maxIndex == J)
271 {
272 // delete /clear the arg vector entry
273 if (dataObj.isParent.first == false)
274 {
275 // Store the key of the element to delete
276 int indexToDelete = std::distance(arg->begin(), it);
277 keysToDelete.push_back(indexToDelete);
278 }
279 else
280 {
281 // clean dataPoints vector
282 dataPoints.clear();
283 }
284 continue;
285 }
286
287 // calculate the coefficients b0, b1, and b2 for the quadratic extrapolation
288 Matrix B = linreg(xDataTemp, yDataTemp, 2);

290 for (int i = 0; i < I; i++)
291 {
292 dataPoints[i]->setY(exp(B(0, 0) + B(1, 0) * (dataPoints[i]->x() - dataPoints[0]->x()) + B(2, 0) * (dataPoints[i]->x() - dataPoints[0]->x()) * (dataPoints[i]->x() - dataPoints[0]->x())));
293 dataPoints[i]->df = 0;
294 // check if the extrapolated value is larger than 3 * MAX
295 if (dataPoints[i]->y() > 3 * MAX)
296 {
297 dataPoints[i]->setY(3 * MAX);
298 }
299 }
300 for (int j = J + 1; j < dataPoints.size(); j++)
301 {
302 dataPoints[j]->setY(exp(B(0, 0) + B(1, 0) * (dataPoints[j]->x() - dataPoints[0]->x()) + B(2, 0) * (dataPoints[j]->x() - dataPoints[0]->x()) * (dataPoints[j]->x() - dataPoints[0]->x())));
303 dataPoints[j]->df = 0;
304 // check if the extrapolated value is larger than 3 * MAX
305 if (dataPoints[j]->y() > 3 * MAX)
306 {
307 dataPoints[j]->setY(3 * MAX);
308 }
309 }
310
311 /* Handle the middle of the data. Hereby, we use a quadratic interpolation to fill the gaps between the data points. */
312 // iterate over the data points from I to J
313 for (int i = I; i <= J; i++)
314 {
315 if (dataPoints[i]->y() == 0.0)
316 {
317 // now, we asume the last value before the zero appears at index "i-1" and we need to find the next data point "j" that is not zero to define the range for the interpolation
318 int j = i + 1;
319 while (dataPoints[j]->y() == 0.0)
320 {
321 if (j >= J)
322 {
323 break;
324 }
325 j++;
326 }
327 // now we have the range from "i" to "j-1" for the interpolation and we need to find the next two data points that are not zero on each side of the range to use them for the interpolation as reference points
328 xDataTemp.clear();
329 yDataTemp.clear();
330 int k = i - 1; // first reference point to the left
331 int l = j; // first reference point to the right
332
333 // side: left
334 // find the next two data points that are not zero
335 int flag = 0;
336 while (flag < 2)
337 {
338 if (k < I)
339 {
340 break;
341 }
342 if (dataPoints[k]->y() > 0.0)
343 {
344 xDataTemp.push_back(dataPoints[k]->x() - dataPoints[0]->x());
345 yDataTemp.push_back(log(dataPoints[k]->y())); // we use log space for interpolation
346 flag++;
347 }
348 k--;
349 }
350 // side: right
351 // find the next two data points that are not zero
352 flag = 0;
353 while (flag < 2)
354 {
355 if (l > J)
356 {
357 break;
358 }
359 if (dataPoints[l]->y() > 0.0)
360 {
361 xDataTemp.push_back(dataPoints[l]->x() - dataPoints[0]->x());
362 yDataTemp.push_back(log(dataPoints[l]->y())); // we use log space for interpolation
363 flag++;
364 }
365 l++;
366 }
367 // calculate the coefficients b0, b1, and b2 for the quadratic interpolation
368 B = linreg(xDataTemp, yDataTemp, 2);
369 // calculate the interpolated y-axis values
370 for (int n = i; n < j; n++)
371 {
372 dataPoints[n]->setY(exp(B(0, 0) + B(1, 0) * (dataPoints[n]->x() - dataPoints[0]->x()) + B(2, 0) * (dataPoints[n]->x() - dataPoints[0]->x()) * (dataPoints[n]->x() - dataPoints[0]->x())));
373 dataPoints[n]->df = 0;
374 // check if the interpolated value is larger than 3 * MAX
375 if (dataPoints[n]->y() > 3 * MAX)
376 {
377 dataPoints[n]->setY(3 * MAX);
378 }
379 }
380 }
381 }
382 }
383 }
384 // Delete the elements after the iteration
385 arg->erase(std::remove_if(arg->begin(), arg->end(),
386 [&keysToDelete, &arg](const auto& item) {
387 // Check if the current index is in keysToDelete
388 return std::find(keysToDelete.begin(), keysToDelete.end(), &item - &arg->front()) != keysToDelete.end();
389 }), arg->end()); },
