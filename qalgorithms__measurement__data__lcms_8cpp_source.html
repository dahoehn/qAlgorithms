28 void LCMSData::readCSV(std::string filename, int rowStart, int rowEnd, int colStart, int colEnd, char separator, std::vector<DataField> variableTypes)
29 {
30 // open the file
31 std::ifstream file(filename);
32 std::string line;
33 std::vector<std::vector<std::string>> raw_data;
34
35 // if rowEnd is -1, then set it to the maximum number of rows
36 if (rowEnd == -1)
37 {
38 rowEnd = 1000000;
39 }
40
41 // if colEnd is -1, then set it to the maximum number of columns
42 if (colEnd == -1)
43 {
44 colEnd = 1000000;
45 }
46
47 // find the DataFiled::SCANNUMBER as this is the key for the LCMSData vector
48 int scanNumberIndex = -1;
49 for (int i = 0; i < variableTypes.size(); i++)
50 {
51 if (variableTypes[i] == DataField::SCANNUMBER)
52 {
53 scanNumberIndex = i;
54 break;
55 }
56 }
57 // check if scanNumberIndex is found
58 if (scanNumberIndex == -1)
59 {
60 std::cerr << "The scan number is not found in the variable types" << std::endl;
61 return;
62 }
63
64 // find the DataFiled::RETENTIONTIME
65 int retentionTimeIndex = -1;
66 for (int i = 0; i < variableTypes.size(); i++)
67 {
68 if (variableTypes[i] == DataField::RETENTIONTIME)
69 {
70 retentionTimeIndex = i;
71 break;
72 }
73 }
74 // check if retentionTimeIndex is found
75 if (retentionTimeIndex == -1)
76 {
77 std::cerr << "The retention time is not found in the variable types" << std::endl;
78 return;
79 }
80
81 // find the DataFiled::MZ
82 int mzIndex = -1;
83 for (int i = 0; i < variableTypes.size(); i++)
84 {
85 if (variableTypes[i] == DataField::MZ)
86 {
87 mzIndex = i;
88 break;
89 }
90 }
91 // check if mzIndex is found
92 if (mzIndex == -1)
93 {
94 std::cerr << "The mz is not found in the variable types" << std::endl;
95 return;
96 }
97
98 // find the DataFiled::INTENSITY
99 int intensityIndex = -1;
100 for (int i = 0; i < variableTypes.size(); i++)
101 {
102 if (variableTypes[i] == DataField::INTENSITY)
103 {
104 intensityIndex = i;
105 break;
106 }
107 }
108 // check if intensityIndex is found
109 if (intensityIndex == -1)
110 {
111 std::cerr << "The intensity is not found in the variable types" << std::endl;
112 return;
113 }
114
115 // read the file
116 int rowCounter = 0;
117 int colCounter = 0;
118 while (std::getline(file, line) && rowCounter < rowEnd)
119 {
120 if (rowCounter >= rowStart)
121 {
122 std::vector<std::string> row;
123 std::stringstream lineStream(line);
124 std::string cell;
125 colCounter = 0;
126 while (std::getline(lineStream, cell, separator))
127 {
128 if (colCounter >= colStart && colCounter < colEnd)
129 {
130 row.push_back(cell);
131 }
132 colCounter++;
133 }
134 raw_data.push_back(row);
135 }
136 rowCounter++;
137 }
138 file.close();
139
140 // check if the raw data has same number of columns as the variableTypes
141 if (raw_data[0].size() != variableTypes.size())
142 {
143 std::cerr << "The number of columns in the raw data does not match the number of variable types" << std::endl;
144 return;
145 }
146
147 // transfere the raw data to the data vector
148 int data_id = 0; // data id is used to identify the data set
149 maxKey = 0;
150 for (int i = 0; i < raw_data.size(); i++)
151 {
152 // check if the scan number is already in the data vector
153 int scanNumber = std::stoi(raw_data[i][scanNumberIndex]);
154 if (this->data.size() < scanNumber+1)
155 { // scan number is not found in the data vector; initialize a new MassSpectrum object and add meta information
156 // add the MassSpectrum object to the data vector
157 this->data.push_back(std::make_unique<MassSpectrum>());
158 // add the scan number to the MassSpectrum object
159 (*this->data[scanNumber]->metaData)[DataField::SCANNUMBER] = VariableType(scanNumber);
160 // add the retention time to the MassSpectrum object
161 (*this->data[scanNumber]->metaData)[DataField::RETENTIONTIME] = VariableType(std::stod(raw_data[i][retentionTimeIndex]));
162 // update the maxKey
163 if (scanNumber > maxKey)
164 {
165 maxKey = scanNumber;
166 }
167 }
168 // create a new DataPoint object and add it to the DataPoint Vector
169 double intensity = std::stod(raw_data[i][intensityIndex]);
170 int df = (intensity > 0) ? 1 : 0;
171 this->data[scanNumber]->dataPoints.push_back(std::make_unique<DataPoint>(intensity, std::stod(raw_data[i][mzIndex]), df));
172 }
173 }

192 //iterate over all data points
