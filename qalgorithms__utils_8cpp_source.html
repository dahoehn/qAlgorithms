102 double erfi(const double x) {
103 /* This function uses the Dawson Integral, i.e.
104 erfi(x) = 2 * Dawson * exp(x^2) / sqrt(pi)
105
106 The Dawson Integral is calculated by Stan Sykora's rational function approximations. URL : http://www.ebyte.it/library/codesnippets/DawsonIntegralApproximations.html (Dawson5())
107 */
108 // calculate the Dawson Integral:
109 double y,p,q;
110 y = x*x;
111 p = 1.0 + y*(0.1049934947 + y*(0.0424060604
112 + y*(0.0072644182 + y*(0.0005064034
113 + y*(0.0001789971)))));
114 q = 1.0 + y*(0.7715471019 + y*(0.2909738639
115 + y*(0.0694555761 + y*(0.0140005442
116 + y*(0.0008327945 + 2*0.0001789971*y)))));
117 double D = x*(p/q);
118
119 return D * std::exp(x*x) / SQRTPI_2; // @todo: define M_2_SQRTPI in utils
120 }

132 const Matrix linreg(
133 const std::vector<double>& xData,
134 const std::vector<double>& yData,
135 const int degree) {
136 // Create the matrix X
137 Matrix X(xData.size(), degree + 1);
138 for (size_t i = 0; i < xData.size(); i++) {
139 for (size_t j = 0; j < degree + 1; j++) {
140 X(i, j) = std::pow(xData[i], j);
141 }
142 }
143 // Create the matrix Y
144 Matrix Y(yData.size(), 1);
145 for (size_t i = 0; i < yData.size(); i++) {
146 Y(i, 0) = yData[i];
147 }
148 // Calculate the coefficients
149 Matrix X_T = X.T();
150 Matrix X_T_X = X_T * X;
151 Matrix X_T_X_inv = X_T_X.inv();
152 Matrix X_T_Y = X_T * Y;
153 Matrix coefficients = X_T_X_inv * X_T_Y;
154 return coefficients;
155 }
